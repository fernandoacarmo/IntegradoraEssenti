CREATE OR REPLACE TRIGGER ESG_WMS_SUPERVISOR
 BEFORE
  INSERT OR DELETE OR UPDATE
     OF CODSUPERVISOR,
        NOME,
        CODGERENTE
 ON PCSUPERV
REFERENCING NEW AS NEW OLD AS OLD
 FOR EACH ROW
DECLARE
      vnACAO ESTREPLICACAO_CORPEM.acao%type;
      vsID   ESTREPLICACAO_CORPEM.id%type;
   BEGIN
     vnAcao := 0; --Inserindo

     IF (:NEW.CODSUPERVISOR IS NULL) OR
        (:NEW.NOME IS NULL) OR
        (:NEW.CODGERENTE IS NULL)
        THEN
        vnACAO := -1;
     END IF;

     IF vnACAO > -1 THEN
     BEGIN
       vsID := :NEW.ROWID;
       IF DELETING THEN
          vnACAO := 2; --Excluindo
          vsID := :OLD.ROWID;
       ELSE
          vsID := :NEW.ROWID;
          IF UPDATING THEN
            vnAcao := 1; -- Alterando
          END IF;
       END IF;
       INSERT INTO ESTREPLICACAO_CORPEM (TABELA, ACAO, ID)
           VALUES ('PCSUPERV'
                  , vnACAO
                  , vsID);
      EXCEPTION
        WHEN DUP_VAL_ON_INDEX THEN
          NULL;
       END;
     END IF;
   END;
